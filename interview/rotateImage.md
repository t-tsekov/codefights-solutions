    [[7, 4, 1],
     [8, 5, 2],
     [9, 6, 3]]
</code></pre>

<ul>
<li><strong>[time limit] 4000ms (py3)</strong></li>
</ul>

<li>
<p><strong>[input] array.array.integer a</strong></p>

</li>
